# From common\species_rights\00_citizenship_types.txt#105

# from = country
# this = species
citizenship_assimilation = {
	# New citizenships cannot be created by mod.
	# Modifying existing Assimilation to replace pop with starting species
	token = citizenship_assimilation

	potential = {
		OR = {
			host_has_dlc = "Synthetic Dawn Story Pack"
			host_has_dlc = "Utopia"
		}
		exists = from
		from = {
			NOT = { has_civic = civic_fanatic_purifiers }
			NOT = { has_civic = civic_hive_devouring_swarm }
			OR = {
				AND = { # machine empires
					# Changed: All Machine Empires have the potential to assimilate
					has_authority = auth_machine_intelligence
				}
				AND = { # non-devouring hive mind
					has_authority = auth_hive_mind
					ROOT = { is_robotic_species = no }
				}
				AND = {
					# Changed: All biologically ascended empires can assimilate, even hive minds, except devouring
					has_ascension_perk = ap_evolutionary_mastery
				}
				has_country_flag = synthetic_empire # synthetically ascended empire
				has_country_flag = flesh_weakened # cyborg empire
				has_ascension_perk = ap_transcendence # psionic assimilation
			}
		}
	}

	allow = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = MUST_NOT_BE_DOMINANT_SPECIES
			NOT = { is_exact_same_species = from }
		}
		custom_tooltip = {
			fail_text = "lonely_hive_mind_pops_kill_themselves_citizenship_assimilation"
			if = {
				limit = { is_lonely_hive_mind_pop = yes }
				from = {
					has_ascension_perk = ap_evolutionary_mastery
				}
			}
		}
		custom_tooltip = {
			fail_text = "disconnected_machine_pops_collapse_citizenship"
			is_disconnected_machine_pop = no
		}
		if = {
			# Changed: Machine Empires can now assimilate all robots, and Driven Assimilators can assimilate anyone
			limit = {
				from = {
					has_authority = auth_machine_intelligence 
					NOT = { has_civic = civic_machine_assimilator }
				}
			}
			custom_tooltip = {
				fail_text = MACHINE_SPECIES_MUST_BE_ROBOTIC
				OR = {
					has_trait = trait_mechanical
					has_trait = trait_machine_unit
				}
			}
		}
		else_if = {
			limit = {
				from = { has_authority = auth_hive_mind }
			}
			custom_tooltip = { # non-devouring hive mind
				# Changed: Non-devouring hive minds can now assimilate other hive minds'
				# drones by mutation (but still not robots).
				fail_text = HIVE_SPECIES_MUST_BE_ORGANIC
				from = {
					has_ascension_perk = ap_evolutionary_mastery
				}
				NOR = {
					has_trait = trait_mechanical
					has_trait = trait_machine_unit
				}
			}
		}
		# Changed: Synthetically ascended empire can now assimilate other machines
		
		else_if = { # cyborg empire
			# Changed: since synthetically ascended empires are not handled above,
			# exclude them here
			limit = {
				from = {
					has_country_flag = flesh_weakened
					NOT = { has_country_flag = synthetic_empire }
				}
			}
			custom_tooltip = {
				fail_text = CYBORG_SPECIES_MUST_BE_ORGANIC
				NOR = {
					has_trait = trait_mechanical
					has_trait = trait_machine_unit
					has_trait = trait_cybernetic
				}
				is_robotic_species = no
			}
		}
		else_if = {
			# Changed: Cyborg and psionic species can be assimilated via psionics
			limit = {
				from = { has_ascension_perk = ap_transcendence }
			}
			custom_tooltip = { # species must be partially organic to transcend
				fail_text = SPECIES_MUST_BE_NONROBOTIC_FOR_MUTATIVE_ASSIMILATION
				is_robotic_species = no
			}
			custom_tooltip = { # transcending species cannot be gestalt consciousness
				fail_text = TRANSCENDING_SPECIES_CANT_BE_HIVE
				NOT = { has_trait = trait_hive_mind }
			}
		}
		# Changed: Any nonrobotic species can be assimilated by empires with
		# evolutionary mastery
		else_if = {
			limit = {
				from = { has_ascension_perk = ap_evolutionary_mastery }
			}
			custom_tooltip = {
				fail_text = SPECIES_MUST_BE_NONROBOTIC_FOR_MUTATIVE_ASSIMILATION
				is_robotic_species = no
			}
		}
	}

	ai_will_do = {
		factor = 100
		modifier = {
			factor = 0
			from = {
				OR = {
					has_valid_civic = civic_hive_devouring_swarm
					has_valid_civic = civic_fanatic_purifiers
					has_valid_civic = civic_machine_terminator
				}
			}
		}
	}
}
