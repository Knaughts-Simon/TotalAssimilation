citizenship_assimilation = {
	is_assimilation = yes

	potential = {
		OR = {
			host_has_dlc = "Synthetic Dawn Story Pack"
			host_has_dlc = "Utopia"
		}
		exists = from
		from = {
			is_homicidal = no
			OR = {
				AND = { # machine assimilator
					has_authority = auth_machine_intelligence
					has_civic = civic_machine_assimilator
					# Changed: Allow Driven Assimilator to assimilate cybernetics/robots/machines
				}
				AND = { # machine
					has_authority = auth_machine_intelligence
					has_active_tradition = tr_synthetics_adopt_machine
					# Changed: Allow machine to assimilate non-robots
				}
				# Changed: Allow hive to assimilate robots
				AND = {
					has_tradition = tr_synthetics_synthetic_age # synth assimilation
					NOT = { has_authority = auth_machine_intelligence }
				}
				has_tradition = tr_genetics_creation # Changed: genetic assimilation
				has_tradition = tr_cybernetics_transubstantiation_synthesis # cybernetic assimilation
				has_tradition = tr_psionics_psionic_assimilation # psionic assimilation
			}
		}
	}

	allow = {
		hidden_trigger = { exists = from }
		custom_tooltip = {
			fail_text = MUST_NOT_BE_DOMINANT_SPECIES
			NOT = { is_exact_same_species = from }
		}
	}

	ai_will_do = {
		factor = 100
		modifier = {
			factor = 0
			from = {
				OR = {
					has_valid_civic = civic_hive_devouring_swarm
					has_valid_civic = civic_fanatic_purifiers
					has_valid_civic = civic_machine_terminator
				}
			}
		}
	}
}
