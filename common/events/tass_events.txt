# From common\events\on_action_events_1.txt#7949
# Total Assimilate Pops on Yearly Pulse

namespace = tass

event = {
	id = tass.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		any_playable_country = {
			num_unique_species > 1
			any_owned_species = {
				has_citizenship_type = {
					type = citizenship_total_assimilation
					country = prev
				}
			}
		}
	}

	immediate = {
		log = "tass.1"
		set_update_modifiers_batch = begin
		every_playable_country = {
			limit = {
				num_unique_species > 1
				any_owned_species = {
					has_citizenship_type = {
						type = citizenship_total_assimilation
						country = prev
					}
				}
			}
			log = "tass.1.epc"

			# Changed: No need to create a new species

			every_owned_planet = {
				planet_event = { id = tass.2 }
			}
		}
		set_update_modifiers_batch = end
	}
}

# Total Assimilate Pops
planet_event = {
	id = tass.2
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		any_owned_species = {
			has_citizenship_type = {
				type = citizenship_total_assimilation
				country = root.owner
			}
		}
	}

	immediate = {
		random_list = {
			1 = { # 3 pops
				set_variable = {
					which = total_assimilation_counter
					value = 3
				}
				while = {
					count = 3
					limit = {
						any_owned_pop = {
							has_citizenship_type = {
								type = citizenship_total_assimilation
								country = root.owner
							}
						}
						check_variable = {
							which = total_assimilation_counter
							value > 0
						}
					}
					total_assimilation_effect = yes
				}
			}

			1 = { # 6 pops
				set_variable = {
					which = total_assimilation_counter
					value = 6
				}
				while = {
					count = 6
					limit = {
						any_owned_pop = {
							has_citizenship_type = {
								type = citizenship_total_assimilation
								country = root.owner
							}
						}
						check_variable = {
							which = total_assimilation_counter
							value > 0
						}
					}
					total_assimilation_effect = yes
				}
			}

			1 = { # 12 pops
				set_variable = {
					which = total_assimilation_counter
					value = 12
				}
				while = {
					count = 12
					limit = {
						any_owned_pop = {
							has_citizenship_type = {
								type = citizenship_total_assimilation
								country = root.owner
							}
						}
						check_variable = {
							which = total_assimilation_counter
							value > 0
						}
					}
					total_assimilation_effect = yes
				}
			}
		}
		set_variable = {
			which = total_assimilation_counter
			value = 0
		}
		if = {
			limit = {
				owner = {
					has_tradition = tr_cybernetics_assimilator_inevitable_assimilation
				}
			}
			set_variable = {
				which = total_assimilation_counter
				value = 1
			}
			while = {
				count = 1
				limit = {
					any_owned_pop = {
						has_citizenship_type = {
							type = citizenship_total_assimilation
							country = root.owner
						}
					}
					check_variable = {
						which = total_assimilation_counter
						value > 0
					}
				}
				total_assimilation_effect = yes
			}
			set_variable = {
				which = total_assimilation_counter
				value = 0
			}
		}
	}
}
